import React, { useState } from 'react';

export default function AddProduct() {
  const [selectedType, setSelectedType] = useState('');
  const descriptions = {
    "Unique": "Des appareils comme téléphones, ordinateurs, etc.",
    "Variable": "T-shirts, pantalons, chaussures, etc.",
    "External/Affiliate product": "Produits comestibles, boissons, etc.",
    "Autres": "Autres types de produits."
  };

  return (
    <div className="p-8">
      <h2 className="text-3xl font-bold text-[#f6858b] mb-4">Ajouter un nouveau produit</h2>
      <form action="">
        <div className="p-4 bg-white rounded shadow">
          <div className="flex gap-4">
            <div className="w-2/3 p-4">
              <div className="mb-4">
                <label className="block mb-2 font-semibold">Titre</label>
                <input
                  type="text"
                  className="w-full border border-gray-300 rounded px-2 py-1"
                  placeholder="Nom du produit..."
                />
              </div>
              <div>
                <label
                  className="flex items-center mb-2 font-semibold gap-2 cursor-pointer"
                  title="Choisissez un type pour afficher sa description"
                >
                  <span>Type de produit</span>
                  <span className="w-5 h-5 bg-black text-white text-xs font-bold rounded-full flex items-center justify-center">
                    ?
                  </span>
                </label>
                <select
                  className="w-full border border-gray-300 rounded p-2"
                  value={selectedType}
                  onChange={(e) => setSelectedType(e.target.value)}
                >
                  <option value="" disabled>
                    -- Sélectionnez un type --
                  </option>
                  {Object.keys(descriptions).map((type) => (
                    <option key={type} value={type}>
                      {type}
                    </option>
                  ))}
                </select>

                {/* Contenu dynamique */}
                {selectedType && (
                  <div className="mt-4 p-3 bg-gray-100 border border-gray-300 rounded">
                    {descriptions[selectedType]}
                  </div>
                )}
              </div>
            </div>

            <div className="w-1/3 p-4 border-4 border-dotted border-gray-500 rounded font-bold">
              Contenu 2
            </div>
          </div>
        </div>
      </form>
    </div>
  );
}
